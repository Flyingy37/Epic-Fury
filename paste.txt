// ─── TABS ────────────────────────────────────────────────────
function renderTabs() {
  const nav = document.getElementById('tab-nav');
  nav.innerHTML = '';
  sections.forEach((s,i) => {
    const btn = document.createElement('button');
    btn.className = `tab-btn flex items-center gap-1.5 px-3 py-2 rounded-lg text-xs font-medium whitespace-nowrap ${i===currentSection?'active':''}`;
    btn.style.background = i===currentSection ? '#f97316' : '#1c0f07';
    btn.style.color      = i===currentSection ? '#fff'    : '#fdba74';
    btn.style.border     = i===currentSection ? 'none'    : '1px solid #431407';
    btn.innerHTML = `<i data-lucide="${s.icon}" style="width:14px;height:14px;"></i> ${s.shortTitle}`;
    btn.onclick = () => goTo(i);
    nav.appendChild(btn);
  });
  lucide.createIcons();
}
function goTo(idx) {
  currentSection = idx;
  renderTabs(); renderContent(); updateProgress();
}
function updateProgress() {
  const pct = Math.round(((currentSection+1)/sections.length)*100);
  document.getElementById('progress-bar').style.width = pct+'%';

}

function renderContent() {
  const area = document.getElementById('content-area');

    case 'data':      html = renderData();      break;

    case 'serial':    html = renderSerial();    break;

    <div class="section-card rounded-2xl border p-5 sm:p-8 mb-4" style="background:#100c08;border-color:#2a1810;animation-delay:0.05s;">

        </div>
        <h2 class="text-lg sm:text-xl font-bold text-white">${s.title}</h2>
      </div>
      ${html}
    </div>
    ${renderNav()}
  `;

}
function renderNav() {
  const prev = currentSection > 0
    ? `<button onclick="goTo(${currentSection-1})" class="flex items-center gap-2 px-5 py-2.5 rounded-xl font-medium text-sm text-white" style="background:#1c0f07;border:1px solid #431407;"><i data-lucide="arrow-right" style="width:16px;height:16px;"></i> הקודם</button>`
    : '<div></div>';
  const next = currentSection < sections.length-1
    ? `<button onclick="goTo(${currentSection+1})" class="flex items-center gap-2 px-5 py-2.5 rounded-xl font-medium text-sm text-white" style="background:#f97316;"><span>הבא</span> <i data-lucide="arrow-left" style="width:16px;height:16px;"></i></button>`
    : '<div></div>';

}
